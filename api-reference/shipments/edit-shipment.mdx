---
title: "Edit Shipment"
api: "PUT https://api.evership.app/v1/shipments/{shipment_id}"
description: "Update an existing shipment's details"
---

Updates the details of an existing shipment. You can modify the sender, receiver, parcels, and other shipment properties.

## Path Parameters

<ParamField path="shipment_id" type="string" required>
  The ID of the shipment to edit.
</ParamField>

## Request Body

<ParamField body="shipment_title" type="string">
  Updated reference title for the shipment.
</ParamField>

<ParamField body="shipment_type" type="string">
  Updated shipment type. Possible values: `outbound`, `inbound`, `transfer`.
</ParamField>

<ParamField body="sender" type="object">
  Updated sender details.
  <Expandable title="properties">
    <ParamField body="name" type="string">Sender's full name</ParamField>
    <ParamField body="contact" type="string">Sender's phone number</ParamField>
    <ParamField body="email" type="string">Sender's email address</ParamField>
    <ParamField body="collection_address" type="object">
      <Expandable title="properties">
        <ParamField body="address_1" type="string">Primary street address</ParamField>
        <ParamField body="address_2" type="string">Secondary address line</ParamField>
        <ParamField body="suburb" type="string">Suburb name</ParamField>
        <ParamField body="city" type="string">City name</ParamField>
        <ParamField body="province" type="string">Province name</ParamField>
        <ParamField body="postal_code" type="string">Postal code</ParamField>
        <ParamField body="country" type="string">Country name</ParamField>
        <ParamField body="latitude" type="number">Latitude coordinate</ParamField>
        <ParamField body="longitude" type="number">Longitude coordinate</ParamField>
      </Expandable>
    </ParamField>
    <ParamField body="pickup_instructions" type="string">Instructions for the driver at pickup</ParamField>
  </Expandable>
</ParamField>

<ParamField body="receiver" type="object">
  Updated receiver details.
  <Expandable title="properties">
    <ParamField body="name" type="string">Receiver's full name</ParamField>
    <ParamField body="contact" type="string">Receiver's phone number</ParamField>
    <ParamField body="email" type="string">Receiver's email address</ParamField>
    <ParamField body="delivery_address" type="object">
      <Expandable title="properties">
        <ParamField body="address_1" type="string">Primary street address</ParamField>
        <ParamField body="address_2" type="string">Secondary address line</ParamField>
        <ParamField body="suburb" type="string">Suburb name</ParamField>
        <ParamField body="city" type="string">City name</ParamField>
        <ParamField body="province" type="string">Province name</ParamField>
        <ParamField body="postal_code" type="string">Postal code</ParamField>
        <ParamField body="country" type="string">Country name</ParamField>
        <ParamField body="latitude" type="number">Latitude coordinate</ParamField>
        <ParamField body="longitude" type="number">Longitude coordinate</ParamField>
      </Expandable>
    </ParamField>
    <ParamField body="drop_off_instructions" type="string">Instructions for the driver at delivery</ParamField>
  </Expandable>
</ParamField>

<ParamField body="parcels" type="array">
  Updated array of parcels.
  <Expandable title="properties">
    <ParamField body="reference" type="string">Parcel reference number</ParamField>
    <ParamField body="description" type="string">Description of parcel contents</ParamField>
    <ParamField body="dimensions" type="object">
      <Expandable title="properties">
        <ParamField body="length_cm" type="number">Length in cm</ParamField>
        <ParamField body="width_cm" type="number">Width in cm</ParamField>
        <ParamField body="height_cm" type="number">Height in cm</ParamField>
      </Expandable>
    </ParamField>
    <ParamField body="weight_kg" type="number">Weight in kg</ParamField>
  </Expandable>
</ParamField>

<ParamField body="carrier_service" type="string">
  Updated carrier service (e.g. "fw-brown").
</ParamField>
