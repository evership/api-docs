---
title: "Create Shipment"
api: "POST https://api.evership.app/v1/shipments"
description: "Create a new shipment with sender, receiver, and parcel details"
---

Creates a new shipment. You can optionally purchase a label immediately by setting `purchase_label` to `true`.

## Request Body

<ParamField body="shipment_title" type="string">
  A reference title for the shipment (e.g. order number).
</ParamField>

<ParamField body="shipment_type" type="string" required>
  The type of shipment. Possible values: `outbound`, `inbound`, `transfer`.
</ParamField>

<ParamField body="sender" type="object" required>
  Sender details and collection address.
  <Expandable title="properties">
    <ParamField body="name" type="string" required>Sender's full name</ParamField>
    <ParamField body="contact" type="string" required>Sender's phone number</ParamField>
    <ParamField body="email" type="string" required>Sender's email address</ParamField>
    <ParamField body="collection_address" type="object" required>
      <Expandable title="properties">
        <ParamField body="address_1" type="string" required>Primary street address</ParamField>
        <ParamField body="address_2" type="string">Secondary address line</ParamField>
        <ParamField body="suburb" type="string" required>Suburb name</ParamField>
        <ParamField body="city" type="string" required>City name</ParamField>
        <ParamField body="province" type="string" required>Province name</ParamField>
        <ParamField body="postal_code" type="string" required>Postal code</ParamField>
        <ParamField body="country" type="string" required>Country name</ParamField>
        <ParamField body="latitude" type="string">Latitude coordinate (optional)</ParamField>
        <ParamField body="longitude" type="string">Longitude coordinate (optional)</ParamField>
      </Expandable>
    </ParamField>
    <ParamField body="pickup_instructions" type="string">Instructions for the driver at pickup</ParamField>
  </Expandable>
</ParamField>

<ParamField body="receiver" type="object" required>
  Receiver details and delivery address.
  <Expandable title="properties">
    <ParamField body="name" type="string" required>Receiver's full name</ParamField>
    <ParamField body="contact" type="string" required>Receiver's phone number</ParamField>
    <ParamField body="email" type="string" required>Receiver's email address</ParamField>
    <ParamField body="delivery_address" type="object" required>
      <Expandable title="properties">
        <ParamField body="address_1" type="string" required>Primary street address</ParamField>
        <ParamField body="address_2" type="string">Secondary address line</ParamField>
        <ParamField body="suburb" type="string" required>Suburb name</ParamField>
        <ParamField body="city" type="string" required>City name</ParamField>
        <ParamField body="province" type="string" required>Province name</ParamField>
        <ParamField body="postal_code" type="string" required>Postal code</ParamField>
        <ParamField body="country" type="string" required>Country name</ParamField>
        <ParamField body="latitude" type="string">Latitude coordinate (optional)</ParamField>
        <ParamField body="longitude" type="string">Longitude coordinate (optional)</ParamField>
      </Expandable>
    </ParamField>
    <ParamField body="drop_off_instructions" type="string">Instructions for the driver at delivery</ParamField>
  </Expandable>
</ParamField>

<ParamField body="parcels" type="array" required>
  Array of parcels in the shipment.
  <Expandable title="properties">
    <ParamField body="reference" type="string">Parcel reference number</ParamField>
    <ParamField body="description" type="string">Description of parcel contents</ParamField>
    <ParamField body="dimensions" type="object" required>
      <Expandable title="properties">
        <ParamField body="length_cm" type="number" required>Length in cm</ParamField>
        <ParamField body="width_cm" type="number" required>Width in cm</ParamField>
        <ParamField body="height_cm" type="number" required>Height in cm</ParamField>
      </Expandable>
    </ParamField>
    <ParamField body="weight_kg" type="number" required>Weight in kg</ParamField>
  </Expandable>
</ParamField>

<ParamField body="carrier_service" type="string">
  Specific carrier service to use. Leave empty to choose later.
</ParamField>

<ParamField body="purchase_label" type="boolean" default="false">
  When `true`, immediately purchases a label for the shipment.
</ParamField>

<ParamField body="collection_date" type="string">
  Requested collection date in `YYYY-MM-DD` format.
</ParamField>

<ParamField body="collection_window" type="string">
  Collection time window. Example: `anytime`.
</ParamField>

<ParamField body="label_format" type="string">
  Label format to use when `purchase_label` is `true`. Possible values: `10_10`, `10_15`, `a4_page`.
</ParamField>
