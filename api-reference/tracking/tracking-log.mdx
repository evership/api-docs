---
title: "Tracking Log"
api: "GET https://api.evership.app/v1/shipments/tracking-log"
description: "Get the detailed tracking log for a shipment"
---

Returns the full tracking history for a shipment. You can look up by either shipment ID or waybill number.

## Query Parameters

<ParamField query="shipment_id" type="string">
  The shipment ID to get the tracking log for.
</ParamField>

<ParamField query="waybill_number" type="string">
  The carrier waybill/tracking number to look up.
</ParamField>

<Note>Provide either `shipment_id` or `waybill_number` — at least one is required.</Note>

<RequestExample>
```bash By shipment ID
curl -X GET "https://api.evership.app/v1/shipments/tracking-log?shipment_id=SHIPMENT_ID" \
  -H "apiKey: YOUR_API_KEY"
```

```bash By waybill number
curl -X GET "https://api.evership.app/v1/shipments/tracking-log?waybill_number=CTR794257438" \
  -H "apiKey: YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch('https://api.evership.app/v1/shipments/tracking-log?waybill_number=CTR794257438', {
  method: 'GET',
  headers: {
    'apiKey': 'YOUR_API_KEY'
  }
});
const data = await response.json();
```

```typescript TypeScript
interface TrackingLogEntry {
  timestamp: string;
  primary_status: {
    status: string;
    slug: string;
    code: number | null;
  };
  comments: string;
  location: string;
}

interface TrackingLogResponse {
  shipment_id: string;
  shipment_title: string;
  waybill_number: string;
  latest_status: {
    primary_status: {
      status: string;
      slug: string;
      code: number;
    };
    secondary_status: string;
  };
  created_time: string;
  last_updated_time: string;
  est_delivery: string;
  est_delivery_text: string;
  tracking_log: TrackingLogEntry[];
}

const waybillNumber: string = 'CTR794257438';

const response = await fetch(
  `https://api.evership.app/v1/shipments/tracking-log?waybill_number=${waybillNumber}`,
  {
    method: 'GET',
    headers: {
      'apiKey': 'YOUR_API_KEY'
    }
  }
);
const data: TrackingLogResponse = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.evership.app/v1/shipments/tracking-log',
    headers={'apiKey': 'YOUR_API_KEY'},
    params={'waybill_number': 'CTR794257438'}
)
data = response.json()
```
