---
title: "Tracking Status"
api: "GET https://api.evership.app/v1/shipments/tracking-status"
description: "Get the current tracking status for one or more shipments"
---

Returns the current tracking status for one or more shipments. Supports batch lookups with comma-separated IDs.

## Query Parameters

<ParamField query="shipment_ids" type="string" required>
  Comma-separated list of shipment IDs to get tracking status for.
</ParamField>

<RequestExample>
```bash cURL
curl -X GET "https://api.evership.app/v1/shipments/tracking-status?shipment_ids=ID_1,ID_2,ID_3" \
  -H "apiKey: YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch('https://api.evership.app/v1/shipments/tracking-status?shipment_ids=ID_1,ID_2,ID_3', {
  method: 'GET',
  headers: {
    'apiKey': 'YOUR_API_KEY'
  }
});
const data = await response.json();
```

```typescript TypeScript
interface TrackingStatusEntry {
  shipment_id: string;
  shipment_title: string;
  waybill_number: string;
  timestamp: string;
  comments: string;
  location: string;
  latest_status: {
    primary_status: {
      status: string;
      slug: string;
      code: number;
    };
    secondary_status: string;
  };
  created_time: string;
  last_updated_time: string;
  est_delivery: string;
  est_delivery_text: string;
}

interface TrackingStatusResponse {
  shipments: TrackingStatusEntry[];
}

const shipmentIds: string[] = ['ID_1', 'ID_2', 'ID_3'];

const response = await fetch(
  `https://api.evership.app/v1/shipments/tracking-status?shipment_ids=${shipmentIds.join(',')}`,
  {
    method: 'GET',
    headers: {
      'apiKey': 'YOUR_API_KEY'
    }
  }
);
const data: TrackingStatusResponse = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.evership.app/v1/shipments/tracking-status',
    headers={'apiKey': 'YOUR_API_KEY'},
    params={'shipment_ids': 'ID_1,ID_2,ID_3'}
)
data = response.json()
```
